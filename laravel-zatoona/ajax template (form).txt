<meta name="csrf-token" content="{{ csrf_token() }}" />



function updatepasswordform(e) {
e.preventDefault();
var data = $('#updatepassword')[0] ;
var formdata = new FormData (data);
$.ajax({
headers:{
	'x-csrf-token':$('meta[name="csrf-token"]').attr('content')
},
url  : "/vendor/updatepassword" ,
type : "POST" , 
data : formdata , 
processData:false ,
contentType:false , 
			
success : function(response) {
	window.location.href = response.url ;
} ,

error :function(response){
let formerrors  = response.responseJSON.errors ;
if(formerrors.hasOwnProperty('old_password')){
$('#old_password_error').html(formerrors.old_password[0]);
}
if(formerrors.hasOwnProperty('new_password')){
$('#new_password_error').html(formerrors.new_password[0]);
}
}

});
}
